// Driver
#define ST7789_DRIVER      // Adafruit 2.0" 320x240 IPS uses this

// Resolution
#define TFT_WIDTH  240     // ST7789 is natively 240x320
#define TFT_HEIGHT 320     // We rotate it to 320x240 in code

// Pins (Matched to your PCB Pinout)
#define TFT_MOSI 23
#define TFT_SCLK 18
#define TFT_CS    5
#define TFT_DC    2
#define TFT_RST   4
#define TFT_BL   15  

// Color/IPS Settings
// Adafruit IPS displays often require Inversion to be ON for black to look black.
// If colors look "negative", comment/uncomment this line.
#define TFT_INVERSION_ON

// If specific colors are swapped (Blue is Red), swap these:
#define TFT_RGB_ORDER TFT_RGB  // Try TFT_BGR if colors are swapped

// Fonts
#define LOAD_GLCD
#define LOAD_FONT2
#define LOAD_FONT4
#define LOAD_FONT6
#define LOAD_FONT7
#define LOAD_FONT8
#define LOAD_GFXFF
#define SMOOTH_FONT

// SPI Speed
#define SPI_FREQUENCY  40000000 // 40MHz for fast updates (Adafruit ST7789 handles this well)
